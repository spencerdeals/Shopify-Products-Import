// Page navigation function goToPage(pageNumber) { // Hide all pages document.querySelectorAll('.page').forEach(page => { page.classList.remove('active'); }); // Show selected page document.getElementById(`page${pageNumber}`).classList.add('active'); // Update progress bar updateProgressBar(pageNumber); // Reset state when going back to page 1 if (pageNumber === 1) { resetToFirstPage(); } currentPage = pageNumber; } // Handle back to products button with loading feedback function handleBackToProducts() { // Provide immediate feedback const backBtn = event.target; backBtn.textContent = 'Resetting...'; backBtn.disabled = true; // Small delay for visual feedback setTimeout(() => { goToPage(1); backBtn.textContent = '← Back to Products'; backBtn.disabled = false; }, 300); } // Reset function for going back to page 1 function resetToFirstPage() { // Clear the URL input document.getElementById('productUrls').value = ''; // Clear any error messages document.getElementById('error').classList.remove('active'); document.getElementById('error').textContent = ''; // Clear loading state document.getElementById('loading').classList.remove('active'); // Clear vendor groups document.getElementById('vendorGroups').innerHTML = ''; // Reset data scrapedProducts = []; vendorGroups = {}; finalCalculation = null; confirmedPrices.clear(); // Reset buttons document.getElementById('continueToFinal').disabled = true; document.getElementById('createOrderBtn').disabled = true; document.getElementById('confirmVariants').checked = false; // Clear cost breakdown document.getElementById('costBreakdown').innerHTML = ''; // Focus on the URL input for better UX setTimeout(() => { document.getElementById('productUrls').focus(); }, 100); }
